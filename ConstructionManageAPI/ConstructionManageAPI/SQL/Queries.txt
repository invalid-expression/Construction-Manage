CREATE TABLE ObservationEntity(
	ID INT PRIMARY KEY IDENTITY(1,1),
	Name VARCHAR(100)
); 

CREATE TABLE DatasEntity(
	ID INT PRIMARY KEY IDENTITY(1,1),
	SamplingTime DATETIME,
	ObservationEntityID INT FOREIGN KEY REFERENCES ObservationEntity(ID)
);

CREATE TABLE PropertiesEntity(
	ID INT PRIMARY KEY IDENTITY(1,1),
	[Label] VARCHAR(100),
	[Value] VARCHAR(100),
	DatasID INT FOREIGN KEY REFERENCES DatasEntity(ID)
);
